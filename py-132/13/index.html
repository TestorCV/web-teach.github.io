Мова SQL


  • SELECT pass - Вибрати колонку.

  • FROM table - З таблиці.

  • SELECT * FROM table - Повний вираз. Означає - ВИБРАТИ УСЕ З ТАБЛИЦІ.

  • INSERT INTO table(pass, name) - Вставити в таблицю в рядки pass та name.

  • VALUES(%s,%s) - Значення. %s - означає, що замість них, ми будемо вставляти якісь данні з пайтону. Просто змінну, якщо треба передати одне значення. Кортеж, якщо треба передати декілька значень.

  • INSERT INTO table(pass, name) VALUES(%s,%s) - Повний вираз. Означає - ВСТАВИТИ В ТАБЛИЦЮ В РЯДКИ pass та name ЗНАЧЕННЯ %s та %s, які потім будуть передаватись кодом.

  • cursor.execute("CREATE TABLE customers (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255), address VARCHAR(255))") - створення таблиці

class MainWindow:
  def __init__(self, root):
      self.root = root
      self.w =  self.root.winfo_screenwidth()//2-500
      self.h =  self.root.winfo_screenheight()//2-350
      self.root.geometry(f'1000x700+{self.w}+{self.h}') #задаємо розміри та позицію вікна
      self.root.configure(bg='#C0FFF6') #задаємо фоновий колір для вікна
      self.root.overrideredirect(1) #приховуємо стандартні кнопки керування вікном

      self.ico = PhotoImage(file='ico.gif')
      self.ico_lbl = Label(self.root, image=self.ico, bg='#C0FFF6' )

      self.root.destroy()

def on_mouse_click(event):
  global dif_x, dif_y

  win_position = [int(coord) for coord in root.wm_geometry().split('+')[1:]]
  dif_x = win_position[0]-event.x_root
  dif_y = win_position[1]-event.y_root

def update_position(event):
  root.wm_geometry('+%d+%d' %(event.x_root+dif_x, event.y_root + dif_y))
